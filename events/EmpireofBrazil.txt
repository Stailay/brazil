    id = aveimperio.2
	title = "EMPIRE1NAME"
	desc = "EMPIRE1DESC"
	picture = DIA_DO_FICO_eventPicture

	    NOT = { exists = BRZ }
		NOT = { has_country_flag = dia_do_fico }

	    name = "EMPIRE1OPT2"

		    change_government = colonial_government

		}

	}

			}

	    name = "EMPIRE4OPT1"
		custom_tooltip = "provinciacisplatinaacquired"

		    	}
